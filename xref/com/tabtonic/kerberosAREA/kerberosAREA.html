42  /**
43   * <h4>kerberosAREA</h4>
44   * This is an AREA plugin designed for use within the BMC Remedy Action Request System.
45   * <br>AREA plugins are a used for delegating authentication responsibility to third party tools.  This plugin receives 
46   * authentication credentials and passes them to MIT Kerberos for validation.

48   * @author      Steve Kallestad, <a href="http://www.tabtonic.com/">TabTonic LLC</a>
49   *
50   * @version     0.0.1-SNAPSHOT      
51   */
52  public class kerberosAREA extends AREAPlugin {
53  	/**
54  	 * for log output
55  	 */
56  	final Logger logger = LoggerFactory.getLogger(kerberosAREA.class);
57  	private ARPluginInfo pluginInfo = new ARPluginInfo("kerberosAREA", this);
58  	private static HashMap<String, Integer> trustedIPs = new HashMap<String, Integer>();
59  	/**
60  	 * Constructor
61  	 */
62  	public kerberosAREA(){
63  		
64  	}
65  	
66  	/**
67  	 * The initialize function is called one time on system startup.
68  	 * 
69  	 * It is being utilized to load configuration parameters.
70  	 * 
71  	 * @param context
72  	 * @throws ARException
73  	 */
74  	@Override
75  	public void initialize(ARPluginContext context) throws ARException {
76  		context.logMessage(pluginInfo, ARPluginContext.PLUGIN_LOG_LEVEL_INFO, "Initializing kerberosAREA");
77  
78  
79  		Properties prop = new Properties();
80  		
81  		InputStream in = getClass().getResourceAsStream("/kerberosAREA.config");
82  		try {
83  			prop.load(in);
84  		} catch (IOException e1) {
85  			logger.error("Failed to load configuration file: kerberosAREA.config: {}", e1.getMessage());
86  		}
87  		
88  		//We need to convert the auth.login.config value before passing it to JAAS
89  		String loginConfigFile=prop.getProperty("java.security.auth.login.config");
90  		String loginConfigFileExForm = getClass().getResource(loginConfigFile).toExternalForm();
91  		prop.setProperty("java.security.auth.login.config", loginConfigFileExForm);
92  		
93  		
94  
95  		//iterate over the kerberos properties and add them to the system properties
96  		Enumeration<Object> propenum = prop.keys();
97  		while(propenum.hasMoreElements()){
98  			String property = (String) propenum.nextElement();
99  			if(property.startsWith("java.security") ||
100 			   property.startsWith("sun.security")){
101 				System.setProperty(property, prop.getProperty(property));
102 			} //add any trusted ip addresses to the hashmap
103 			else if(property.startsWith("trusted")) {
104 				trustedIPs.put(prop.getProperty(property), 1);
105 			}
106 		}
107 		
108 		
109 		super.initialize(context);
110 	}
111 	
112 	/**
113 	 * The AR Server will check with the AREA plugin to determine if user information has expired.
114 		 it does this when it receives a second authentication request at least five minutes after the
115 		 previous authentication request.
116 		
117 		 We always return true in this instance, invalidating any existing user cache info.
118 	 * @param context
119 	 * @return
120 	 * @throws ARException
121 	 */
122 	@Override
123 	public boolean areaNeedSync(ARPluginContext context) throws ARException {
124 		 
125 		return true;
126 	}
127 
128 	/**
129 	 * The areaVerifyLogin function is called by Remedy with the following parameters.  
130 	 * 
131 	 * We return an AREAResponse object which will tell remedy if the authentication is a success or a failure.
132 	 * 
133 	 * @param context
134 	 * @param user
135 	 * @param password
136 	 * @param networkAddress
137 	 * @param authString
138 	 * @return
139 	 * @throws ARException
140 	 */
141 	@Override
142 	public AREAResponse areaVerifyLogin(ARPluginContext context, String user,
143 			String password, String networkAddress, String authString) throws ARException {
144 		logger.trace("Verify login for {} from {}", user, networkAddress);
145 
146 		//we need to return an AREAResponse object
147 		AREAResponse response = new AREAResponse();
148 		
149 		//for trusted IPs (i.e. midtier), accept the given login
150 		if(trustedIPs.containsKey(networkAddress)){
151 			response.setLoginStatus(AREAResponse.AREA_LOGIN_SUCCESS);			
152 			return response;
153 		}
154 
155 		
156 		LoginContext lc;
157 		try {
158 			lc = new LoginContext("primaryLoginContext", new UAndPCallbackHandler(user, password.toCharArray()));
159 			//verify login credentials	
160 			lc.login();
161 			response.setLoginStatus(AREAResponse.AREA_LOGIN_SUCCESS);
162 		} catch (LoginException e) {
163 			response.setLoginStatus(AREAResponse.AREA_LOGIN_FAILED);
164 			logger.error("Authorization Failure for user {}", user);
165 		}			
166 		
167 
168 		return response;
169 		
170 	}
